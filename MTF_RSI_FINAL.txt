calc_double_rsi(src, short_len, long_len, short_smooth, long_smooth) =>
    rsiS = ta.ema(ta.rsi(src, short_len), short_smooth)
    rsiL = ta.ema(ta.rsi(src, long_len), long_smooth)
    diff = rsiS - rsiL
    [rsiS, rsiL, diff]

calc_double_stoch(len, smooth) =>
    raw_stoch = ta.stoch(close, high, low, len)
    db_stoch = ta.ema(ta.ema(raw_stoch, smooth), smooth)
    db_stoch

get_rsi_direction(diff, thresh) =>
    diff > thresh ? 1 : diff < -thresh ? -1 : 0

// RSI Background Ribbon
// Consolidation detection: resets after 4 consecutive bars outside range
// Prevents false breakouts that whipsaw back into the range
var int consol_count = 0
var float consol_hi = na
var float consol_lo = na
var int breakout_count = 0

if consol_count == 0 or na(consol_hi)
    consol_hi := high
    consol_lo := low
    consol_count := 1
    breakout_count := 0
else
    new_hi = math.max(consol_hi, high)
    new_lo = math.min(consol_lo, low)
    new_range = new_hi - new_lo
    if new_range >= consol_range_pts
        // Range exceeded — count consecutive bars outside
        breakout_count := breakout_count + 1
        if breakout_count >= 4
            // Confirmed breakout — reset everything
            consol_hi := high
            consol_lo := low
            consol_count := 1
            breakout_count := 0
        else
            // Not confirmed yet — keep tracking but don't reset
            consol_hi := new_hi
            consol_lo := new_lo
            consol_count := consol_count + 1
    else
        // Back inside range — reset breakout counter
        breakout_count := 0
        consol_hi := new_hi
        consol_lo := new_lo
        consol_count := consol_count + 1
